import pandas as pd
import numpy as np
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans

# Load the data
airlines_data = pd.read_excel("/Users/yemaquinn/Documents/YorkU Big Data Analytics Certificate/CDSA 1100-042/Airlines-Reviews-Data.xlsx")

# Convert 'Recommended' to binary (1 for 'yes', 0 for 'no')
airlines_data['Recommended'] = airlines_data['Recommended'].map({'yes': 1, 'no': 0})

# Drop rows with NaNs in 'Recommended'
airlines_data = airlines_data.dropna(subset=['Recommended'])

# Separate features and target
X = airlines_data.drop(columns=['Recommended'])
y = airlines_data['Recommended']

# Convert all categorical columns to numeric codes
categorical_columns = X.select_dtypes(include=['object']).columns
for col in categorical_columns:
    X[col] = pd.Categorical(X[col]).codes

# Select numeric columns
numeric_columns = X.select_dtypes(include=[np.number]).columns
X_numeric = X[numeric_columns]

# Handle missing values for numeric columns
imputer = SimpleImputer(strategy='mean')
X_numeric = imputer.fit_transform(X_numeric)

# Standardize the features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_numeric)

# Fit KMeans
kmeans = KMeans(n_clusters=3, random_state=42)
clusters = kmeans.fit_predict(X_scaled)

# Add the cluster labels to the original DataFrame
airlines_data['Cluster'] = clusters

# Get cluster centroids
centroids = kmeans.cluster_centers_

# Compute feature importance based on centroids
feature_importance = np.abs(centroids).mean(axis=0)
feature_importance_df = pd.DataFrame({
    'Feature': numeric_columns,
    'Importance': feature_importance
}).sort_values(by='Importance', ascending=False)

# Display feature importance
print("\nFeature importance based on K-means centroids:")
print(feature_importance_df)

# Additional check for feature names and their importance
import matplotlib.pyplot as plt

plt.figure(figsize=(12, 8))
plt.barh(feature_importance_df['Feature'], feature_importance_df['Importance'], color='skyblue')
plt.xlabel('Importance')
plt.title('Feature Importance Based on K-means Centroids')
plt.gca().invert_yaxis()
plt.show()
